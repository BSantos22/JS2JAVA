var array = [1,3,2,4,6,7,2,4,5,1];

function max() {
	var max = -1;
	for (var i = 0; i < array.length; i++) {
		if (array[i] > max) {
			max = array[i];
		}
	}
	
	return max;
}

console.log(max());

var array = [1,3,2,4,6,7,2,4,5,1];

function hasBigger(number) {
	for (var i = 0; i < array.length; i++) {
		if (array[i] > number) {
			return true;
		}
	}
	
	return false;
}

console.log(hasBigger(3), hasBigger(6), hasBigger(8));
